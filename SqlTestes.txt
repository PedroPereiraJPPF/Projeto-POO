@Não testado
CREATE TABLE funcionarios
(
    id serial unique,
    nome varchar(50),
    salario numeric(2,1),
    datadeadmissao varchar(30),
    cpf varchar(20),
	senha varchar(50),
    funcao integer,
    primary key(cpf)
);

create table clientes (
	id serial unique,
	nome varchar(50),
	cpf varchar(50),
	
	primary key (cpf)
);

create table telefonesClientes (
	id serial,
	cpfCliente varchar(50),
	numero varchar(30),
	
	primary key(id),
	foreign key(cpfCliente) references clientes(cpf)
);

create table telefonesFuncionarios(
	id serial,
	cpfFuncionario varchar(50),
	numero varchar(30),
	
	primary key(id),
	foreign key(cpfFuncionario) references funcionarios(cpf)
);

create table enderecos(
	id serial,
	cpfCliente varchar(50),
	cpfFuncionario varchar(50),
	rua varchar(30),
	bairro varchar(30),
	cidade varchar(30),
	estado varchar(40),
	cep varchar(10),
	
	primary key(id),
	foreign key(cpfCliente) references clientes(cpf),
	foreign key(cpfFuncionario) references funcionarios(cpf)
);

create table veiculos(
	id serial unique,
	placa varchar(10),
	cor varchar(20),
	modelo varchar(50) not null,
	tipo varchar(15) not null,
	cpfDono varchar(20),
	
	primary key(placa),
	foreign key(cpfDono) references clientes(cpf) on Update Cascade on delete cascade
);

create table servicos (

);

@Não testado
create table pecas (
	id integer,
	nome varchar(50),
	preco float,
	fabricante varchar(50)
);

@Em teste 
create table orcamentos (
	id serial,
	valor float,
	placaVeiculo varchar(10),
	
	primary key(id),
	foreign key(placaVeiculo) references veiculos(placa) on delete cascade on update cascade
);

create table pecas_orcamentos (
	idOrcamento integer,
	idPeca integer,
	
	foreign key(idOrcamento) references orcamentos(id) on delete cascade on update cascade,
	foreign key(idPeca) references pecas(id) on delete cascade on update cascade
);

create table servicos_orcamentos (
	idOrcamento integer,
	idServico integer,
	
	foreign key(idOrcamento) references orcamentos(id) on delete cascade on update cascade,
	foreign key(idServico) references servicos(id) on delete cascade on update cascade
);	
	foreign key(idOrcamento) references orcamentos(id),
	foreign key(idServico) references servicos(id)
);
